数组

数组是多个变量值的集合，数组是Array对象的实例，因此数组可以调用方法

数组是一个引用类型

数组的声明

创建数组

1//使用对象方式创建数组
2let arr1 = new Array(1, "hello", "world");
3console.log(arr1);
4//使用字面量创建数组 推荐做法
5let arr2 = [2, "你好", "世界"];
6console.log(arr2);
7//多维数组创建
8let arr3 = [[1, "Tom"],[2, "Mike", "boy"]];
9console.log(arr3);

  • 数组是引用类型
  • 数组类型有length属性
  • 数组可以设置任何值
  • 若直接设置一个数组的第三个值，会将该数组的第一、第二个值设置为空值

使用Array.of创建数组

使用Array.of与new Array不同是设置只有一个元素的数组时不会创建空元素数组

1let arr4 = Array.of(3);
2console.log(arr4);
3let arr5 = Array.of(1, 2, 3);
4console.log(arr5);
5
6let arr6 = new Array(3);
7console.log(arr6);

Array.from

使用Array.from可以将类数组转换为数组，类数组指包含length属性或可迭代的对象

Array.from()包含两个参数，第一个参数为要转换的数据，第二个参数为类似于map函数的回调方法

 1<body>
 2  <button message="hello"></button>
 3<button message="world"></button>
 4<script>
 5  //Array.from 将类数组转换为数组
 6  //Array.from()包含两个参数 第一个为需要转换的数据 第二个为一个类似map函数的回调
 7  let str = "hello world";
 8	console.log(Array.from(str))
 9
10	//为对象设置 length 属性后也可以转换为数组 但属性字段名要为数值或数值字符串
11	//如果字段名是其它 该字段的值将不会被转换
12	let user = { 
13  	0: "hello",
14  	name: "tom",
15  	"1": 18,
16  	length: 2
17	};
18	console.log(Array.from(user));
19
20	//获取DOM元素将其转换为数组 通过Array.from()的第二个参数 对DOM元素进行操作
21	let btns = document.querySelectorAll('button');
22	console.log(btns);
23	Array.from(btns, (item) => {
24  	item.style.background = 'red';
25	});
26</script>
27</body>

展开语法

数组合并

可以使用展开语法合并数组，使用…可将数组展开为多个值

1//展开语法合并数组
2let a = [1, 2, 3];
3let b = ['a', 'hello', ...a];
4console.log(b);
5
6//也可以使用concat方法合并
7let c = [1, 2, 3];
8let d = a.concat(['a', 'hello']);
9console.log(d);

![Image 5](chap4_ 5.png)

在函数中使用展开语法

在函数参数列表中使用展开语法，来接受任意数量的参数

 1//在函数参数列表中使用展开语法
 2function test1(...args) {
 3  console.log(args);
 4}
 5test1(1, 2, 3, 'a', 'hello');
 6//也可以接受部分参数
 7function test2(idx, ...args) {
 8  console.log(idx, args);
 9}
10test2('idx000', 1, 2, 3, 'a', 'hello');

![Image 6](chap4_ 6.png)

解构赋值

解构是一种更简洁的赋值特性，可以理解为分解一个数据的结构

 1//解构赋值
 2let [name, age] = ['Tom', 18];
 3console.log(name, age);
 4
 5function test1() {
 6  return ['Miki', 22];
 7};
 8let [a, b] = test1();
 9console.log(a, b);
10
11//剩余结构 通过展开语法 用一个变量来接受剩余参数
12let [c, ...d] = ['idx000', 1, 2, 'hello'];
13console.log(c, d);
14
15//只赋值部分变量
16let [, n] = ['joe', 23];
17console.log(n);

管理数组中的元素

基本用法

数组的索引从0开始 可以通过索引改变数组 也可以向数组追加元素

1//通过索引修改元素
2let arr = [1, 'hello', 'world'];
3arr[1] = 'hi';
4console.log(arr);
5//向数组追加元素
6arr[arr.length] = 'Tom';
7console.log(arr);

slice()方法

使用slice方法从数组中截取部分元素组合成新数组（并不会改变原数组），不传第二个参数时，截取到数组的最后元素

4console.log(arr.slice(2));
5console.log(arr.slice());

splice()方法

使用splice方法可以添加、删除、替换数组中的元素，会对原数组造成改变，返回值为删除的元素

join

使用 join 将数组元素连接成字符串

![Image 20](chap4_ 20.png)

split

split 方法用于将字符串分割成数组，类似 join 方法的反函数

1let score = "99,98,78";
2console.log(score.split(","));

![Image 21](chap4_ 21.png)

concat

concat 方法用于连接两个或多个数组，元素是值类型的是复制操作，如果是引用类型还是指向同一对象

1let arr0 = ["hello", "world"];
2let arr1 = [1, 2];
3let arr2 = [{name: "tom"}, {name: "jim"}];
4arr1[0] = 3;
5console.log(arr0.concat(arr1, arr2));
6//arr1中的元素是值类型 而arr2中的元素是引用类型
7arr1[1] = 4;
8arr2[1].name = "mike";

![Image 22](chap4_ 22.png)

也可以使用展开语法实现数组连接

1let arr0 = ["hello", "world"];
2let arr1 = [1, 2];
3let arr2 = [{name: "tom"}, {name: "jim"}];
4console.log([...arr0, ...arr1, ...arr2]);

![Image 23](chap4_ 23.png)

copyWithin

使用 copyWithin 方法从数组中复制一部分到同数组中的另外位置

1//array.copyWithin(target, start, end);
2//target: 必需，复制到指定目标索引位置
3//start: 可选，元素复制的起始位置
4//end: 可选，停止复制的索引位置，默认为array.length，负值表示倒数
5const arr = [1, 2, 3, 4];
6console.log(arr.copyWithin(2, 0, 2));
