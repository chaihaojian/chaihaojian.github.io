运算符与流程控制

一元运算符的前置与后置操作(++n与n++的区别)

两者在不参与其他运算时结果相同，都是n自增；

1let a = 1;
2let b = 1;
3++a;
4console.log(a);
5b++;
6console.log(b);

在参与其他运算时，++n先自增，再进行运算；n++先进行运算，再自增。

1let n = 1;
2let m = 1;
3let f = 2;
4let d = f + ++n;//n先自增，然后运算f + n
5console.log(d);
6console.log(n);
7let e = f + m++;//先运算f + n, 然后n自增
8console.log(e);
9console.log(m);

比较运算符

== 与 ===

a == b，如果a, b不是相同类型的变量，js会自动将两个变量转换成相同类型再进行比较；a === b 则会直接进行比较。

1let a = 1;
2let b = "1";
3console.log(a == b);
4console.log(a >= b);
5console.log(a === b);

逻辑运算符

逻辑与： &&

逻辑或： ||

逻辑非： !

短路运算

1可以表示true；0可以表示false；

1let a = 1, b = 0;
2console.log(a == true)
3console.log(a == false)
4console.log(b == true)
5console.log(b == false)

在逻辑或进行判断时，如 （a || b）：若a为真，则不再判断b，直接返回真；若a为假，则继续判断b的真假；

短路运算在赋值中的妙用

1let a = 3;
2let b;
3let c = 4;
4let f = b || a;//b为空，因此将a的值赋给f
5console.log(f);
6let p = c || a;//c不为空，因此不需要判断a，直接将c的值赋给p
7console.log(p);

while 循环控制

 1function genTable(options = {tr: 5, td: 3}){
 2  document.write(`<table border = "1" width="100%"`);
 3  while(options.tr-- != 0){
 4    let td = options.td;
 5    document.write(`<tr>`);
 6    while(td-- != 0){
 7      document.write(`<td>${td}</td>`);
 8    }
 9    document.write(`</tr>`);
10  }
11  document.write(`</table>`);
12}
13genTable();
14document.write(`<br/>`);
15genTable({tr: 3, td: 5});

do while 循环控制

 1function genStars(row = 5){
 2  let l = 0;
 3  do {
 4    let n = 0;
 5    do {
 6      document.write("*");
 7    } while (++n <= l);
 8    document.write(`<br/>`);
 9  } while(++l < row);
10};
11genStars();
12document.write(`<br/>`);
13genStars(8);

for 循环

 1function genStars(row = 5){
 2  for (let i = 1; i <= row; i++){
 3    for (let n = row - i; n > 0; n--){
 4      document.write(`<span style="color: white">^</span>`);
 5    }
 6    for (let m = i * 2 - 1; m > 0; m--){
 7      document.write(`*`);
 8    }
 9    document.write(`<br/>`);
10  }
11}
12genStars();
13document.write(`<br/>`);
14genStars(8);

continue 与 break

continue：停止当前循环，进行下一次循环；

break：终止当前循环；

如果需要从子循环直接终止更上一级的循环，需要通过标签：

 1pa: for (let i = 1; i <= 10; i++){
 2  ch: for (let j = 1; j <= 10; j++){
 3    if (j % 2 == 0){
 4      console.log(i, j);
 5    }
 6    if (i + j > 10){
 7      break pa;
 8    };
 9  };
10};
